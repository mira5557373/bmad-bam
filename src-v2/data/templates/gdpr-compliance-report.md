```yaml

  access_request:
    automation: full
    format: json_export
    includes: [profile, activity, ai_interactions]
    
  erasure_request:
    automation: semi
    steps:
      - verify_identity
      - check_legal_holds
      - soft_delete
      - queue_hard_delete
      - notify_processors
```
