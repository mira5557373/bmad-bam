### Safety Rules

```yaml
safety_rules:
  - name: harmful_output_detection
    trigger: ai.output.safety_score < {{safety_threshold}}
    action: l1_kill
    auto_recover: false
  
  - name: runaway_agent
    trigger: agent.runtime > {{max_runtime}} OR agent.tokens > {{max_tokens}}
    action: l1_kill
    auto_recover: true
    recovery_delay: {{recovery_delay}}
  
  - name: tenant_abuse_pattern
    trigger: tenant.abuse_score > {{abuse_threshold}}
    action: l2_kill
    auto_recover: false
    escalation: security_team
  
  - name: model_error_spike
    trigger: model.error_rate > {{error_threshold}} for {{error_window}}
```
