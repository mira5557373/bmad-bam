```yaml
metadata_schema:
  type: object
  properties:
    source:
      type: string
      description: Data origin
    annotator:
      type: string
      description: Human annotator ID
    annotation_date:
      type: date
      description: When annotation was made
    confidence:
      type: number
      minimum: 0
      maximum: 1
      description: Annotation confidence
    tags:
      type: array
      items:
        type: string
      description: Classification tags
    difficulty:
      type: string
      enum: [easy, medium, hard]
      description: Task difficulty
```
